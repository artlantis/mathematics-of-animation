### Polygon Generator

```js
function polygon(sides, radius) {
  const theta = 360 / sides;

  return range(sides).map((idx) => {
    return {
      theta: degreesToRadians(theta * idx),
      r: radius,
    };
  });
}
```

```js
function lerp(start, end) {
  const dx = end.x - start.x;
  const dy = end.y - start.y;

  return {
    x: start.x + dx * 0.1,
    y: start.y + dy * 0.1,
  };
}
```
